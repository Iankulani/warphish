# 🦈 WAR PHISH - Makefile

.PHONY: help install run clean test docker-build docker-run security lint format

# Colors
GREEN  := $(shell echo -e "\033[32m")
YELLOW := $(shell echo -e "\033[33m")
RED    := $(shell echo -e "\033[31m")
RESET  := $(shell echo -e "\033[0m")

help: ## Show this help message
	@echo '$(GREEN)🦈 WAR PHISH Commands$(RESET)'
	@echo ''
	@grep -E '^[a-zA-Z_-]+:.*?## .*$$' $(MAKEFILE_LIST) | sort | awk 'BEGIN {FS = ":.*?## "}; {printf "$(GREEN)%-15s$(RESET) %s\n", $$1, $$2}'

install: ## Install all dependencies
	@echo "$(YELLOW)Installing WAR PHISH...$(RESET)"
	@chmod +x setup.sh && ./setup.sh

run: ## Run WAR PHISH
	@echo "$(YELLOW)Starting WAR PHISH...$(RESET)"
	@source venv/bin/activate && python3 warphish.py

dev: ## Run in development mode with auto-reload
	@echo "$(YELLOW)Starting development mode...$(RESET)"
	@source venv/bin/activate && python3 warphish.py --debug

clean: ## Clean temporary files and cache
	@echo "$(YELLOW)Cleaning...$(RESET)"
	@rm -rf __pycache__ .pytest_cache .mypy_cache .coverage htmlcov
	@find . -type d -name "__pycache__" -exec rm -rf {} + 2>/dev/null || true
	@find . -type f -name "*.pyc" -delete

test: ## Run tests
	@echo "$(YELLOW)Running tests...$(RESET)"
	@source venv/bin/activate && pytest tests/ -v --cov=. --cov-report=term-missing

test-unit: ## Run unit tests only
	@echo "$(YELLOW)Running unit tests...$(RESET)"
	@source venv/bin/activate && pytest tests/test_*.py -v -m "not integration"

test-integration: ## Run integration tests
	@echo "$(YELLOW)Running integration tests...$(RESET)"
	@source venv/bin/activate && pytest tests/test_integration.py -v

security: ## Run security checks
	@echo "$(YELLOW)Running security checks...$(RESET)"
	@source venv/bin/activate && bandit -r warphish.py -lll
	@source venv/bin/activate && safety check -r requirements.txt
	@source venv/bin/activate && pip-audit

lint: ## Run linters
	@echo "$(YELLOW)Running linters...$(RESET)"
	@source venv/bin/activate && flake8 warphish.py --max-line-length=120
	@source venv/bin/activate && pylint warphish.py --exit-zero
	@source venv/bin/activate && mypy warphish.py --ignore-missing-imports

format: ## Format code
	@echo "$(YELLOW)Formatting code...$(RESET)"
	@source venv/bin/activate && black warphish.py
	@source venv/bin/activate && isort warphish.py

docker-build: ## Build Docker image
	@echo "$(YELLOW)Building Docker image...$(RESET)"
	@docker build -t warphish:latest .

docker-run: ## Run Docker container
	@echo "$(YELLOW)Running Docker container...$(RESET)"
	@docker-compose up -d

docker-stop: ## Stop Docker container
	@echo "$(YELLOW)Stopping Docker container...$(RESET)"
	@docker-compose down

docker-logs: ## View Docker logs
	@docker-compose logs -f

db-init: ## Initialize database
	@echo "$(YELLOW)Initializing database...$(RESET)"
	@source venv/bin/activate && python3 -c "from warphish import DatabaseManager; DatabaseManager()"

db-clean: ## Clean database
	@echo "$(YELLOW)Cleaning database...$(RESET)"
	@rm -f .warphish/warphish.db

reports: ## Generate reports directory
	@mkdir -p reports

install-system: ## Install system dependencies (Linux only)
	@echo "$(YELLOW)Installing system dependencies...$(RESET)"
	@sudo apt-get update && sudo apt-get install -y nmap hping3 dsniff macchanger tcpdump netcat-openbsd whois dnsutils curl wget ssh

dev-setup: install ## Full development setup
	@echo "$(YELLOW)Setting up development environment...$(RESET)"
	@pip install -r requirements-dev.txt
	@pre-commit install

check: lint test security ## Run all checks
	@echo "$(GREEN)All checks passed!$(RESET)"

.DEFAULT_GOAL := help
